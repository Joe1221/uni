\section{Der mehrdimensionale Mittelwertsatz}

Alle vorkommenden Funktionen seien stetig differenzierbar.

Sei $f : \R \to \R$, $a,b \in \R$, $b>a$, dann existiert $\xi \in [a,b]$ mit
\[
	\f {f(b)-f(a)}{b-a} = f'(\xi)
\]
Die am häufigsten in der Numerik verwendete Folgerung ist
\begin{align*}
	|f(b) - f(a)| \le |b-a| \sup_{\xi \in [a,b]} |f'(\xi)|
\end{align*}
Der Mittelwertsatz lässt sich nicht direkt ins Mehrdimensionale übertragen.

\begin{lem}[Mittelwertsatz im Mehrdimensionalen]
	Für $F: \R^n \to \R^n$ und $x,y \in \R^n$ gilt
	\[
		F(y) - F(x) &= \int_0^1 F'(x+t(y-x))(y-x) \dx[t]
		&= \int_0^1 F'(x+t(y-x)) \dx[t] (y-x)
	\]
	(das Integral verstehen wir dabei immer komponentenweise).
	Insbesondere folgt
	\begin{align*}
		\|F(y) - F(x)\| \le \|y-x\| \sup_{t\in[0,1]} \|F'(x+t(y-x))\|
	\end{align*}
	\begin{proof}
		Man erinnere sich an die mehrdimensionale Kettenregel und den Hauptsatz der Differential- und Integralrechnung
		\begin{enumerate}[a)]
			\item
				Für $g: \R^s \to \R^n, h: \R^n \to \R^m$ gilt
				\begin{align*}
					(h\circ g)'(x)  = \underbrace{h'(g(x))}_{\in \R^{m\times n}}\underbrace{g'(x)}_{\in \R^{n\times s}}.
				\end{align*}
			\item
				Für $f: \R \to \R$, $x,y\in \R$ gilt
				\[
					f(y) - f(x) = \int_x^y f'(t) \dx[t]
				\]
				Außerdem für $f: \R \to \R^m$:
				\begin{align*}
					f(y) - f(x) = \begin{pmatrix}
						f_1(y) - f_1(x) \\
						\vdots \\
						f_m(y) - f_m(x) 
					\end{pmatrix} = \begin{pmatrix}
						\int_x^y f_1'(t) \dx[t] \\
						\vdots \\
						\int_x^y f_1'(t) \dx[t]
					\end{pmatrix}
					= \int_x^y f'(t) \dx[t]
				\end{align*}
		\end{enumerate}
		Sei $F: \R^n \to \R^n$, $x,y \in \R^n$.
		Definiere $f: \R \to \R^n$ durch
		\[
			f(t) := F(x + t(y-x))
		\]
		Offenbar ist
		\[
			f(t) = F(g(t)) \qquad \text{mit} \quad g(t) := x + t(y-x)
		\]
		Es ist
		\[
			g'(t) = \begin{pmatrix}
				g_1'(t) & \cdots & g_n'(t)
			\end{pmatrix} = y-x
		\]
		Also
		\[
			F(y) - F(x) 
			&= f(1) - f(0) \\
			&= \int_0^1 f'(t) \dx[t] \\
			&= \int_0^1 F'(g(t))g'(t) \dx[t] \\
			&= \int_0^1 F'(x+t(y-x))(y-x) \dx[t].
		\]
	\end{proof}
\end{lem}
